

const range = dv.el("input", "");
range.type = "range";
range.className = "custom-slider";
range.value = 40; // 初始值

// 动态更新颜色的逻辑
const updateColor = (val) => {
    range.style.setProperty('--slider-percent', val + '%');
}

updateColor(range.value);

range.oninput = (e) => {
    updateColor(e.target.value);
}
